/**
 * Timer object returned by setIntervalAsync.<br>
 * Can be used together with {@link clearIntervalAsync} to stop execution.
 */
class SetIntervalAsyncTimer {
  constructor () {
    this.stopped = false
    this.id = 0
    this.timeouts = {}
    this.promises = {}
  }
}

export default SetIntervalAsyncTimer
